# App to retain employee details and select a day of annual leave

import datetime


class Employee:
    """Name | Surname | Employee No | Age | Annual Leave Day"""

    def __init__(self, name, last_name, empno, age):
        self._name = name
        self._surname = last_name
        self._empno = empno
        self.age = age
        self.day = None

    def leave(self):

        """Check if there's leave time booked. If so, run change_leave function"""

        if self.day is None:

            date = input(f"Please {self._name} enter a date (DD-MM-YYYY) to go on leave: \n")

            try:

                self.day = datetime.datetime.strptime(date, "%d-%m-%Y")

            except ValueError:

                print("Invalid date format. Please use DD-MM-YYYY format.")

        else:

            Employee.change_leave(self)

    def change_leave(self):

        """Check if there's leave time booked. if not, run leave function"""

        if self.day is not None:

            answer = input("You already booked leave time. Would you like to change it? Yes or No")

            if answer.lower() == "yes":

                date = input(f"Please {self._name} enter a date (DD-MM-YYYY) to go on leave: \n")

                self.day = datetime.datetime.strptime(date, "%d-%m-%Y")

                print("Leave time successfully booked")

            else:

                print("Thanks and goodbye")

        else:

            Employee.leave(self)

Here's a real application, with a program that checks if a rectangle is contained within a specified circle:

									import math
class Point:

    def __init__(self, x=0, y=0):
        self.x = x
        self.y = y


class Rectangle:

    def __init__(self):
        self.width = 0
        self.height = 0
        self.center = Point()


class Circle:

    def __init__(self):
        self.radius = 0
        self.center = Point()


def point_in_circle(object1, object2):
    """Object1 Circle || Object2 Point"""

    distance = math.sqrt((object1.center.x - object2.x) ** 2 + (object1.center.y - object2.y) ** 2)

    if distance <= object1.radius:
        return True
    else:
        return False


def rect_in_circle(object1, object2):
    """Object1 Circle || Object2 Rectangle"""

    point2 = Point(object2.center.x + object2.width, object2.center.y)
    point3 = Point(object2.center.x, object2.center.y + object2.width)
    point4 = Point(object2.center.x + object2.width, object2.center.y + object2.width)

    points = [object2.center, point2, point3, point4]

    for n in points:
        if not point_in_circle(object1, n):
            return False
    return True


def rect_circle_overlap(object1, object2):
    """Object1 Circle || Object2 Rectangle"""

    point2 = Point(object2.center.x + object2.width, object2.center.y)
    point3 = Point(object2.center.x, object2.center.y + object2.width)
    point4 = Point(object2.center.x + object2.width, object2.center.y + object2.width)

    points = [object2.center, point2, point3, point4]

    for n in points:
        if point_in_circle(object1, n):
            return True
    return False


def main():
    box = Rectangle()
    box.width = 100.0
    box.height = 200.0
    box.corner = Point()
    box.corner.x = 50.0
    box.corner.y = 50.0

    circle = Circle()
    circle.center = Point()
    circle.center.x = 150.0
    circle.center.y = 100.0
    circle.radius = 75.0

    print(point_in_circle(circle, box.center))
    print(rect_in_circle(circle, box))
    print(rect_circle_overlap(circle, box))


if __name__ == '__main__':
    main()

Animal polymorphism

There are many animals. Many of them make sounds, but not all the same sound. Learned:

  • Dynamic polymorphism, as depdending on the type of object passed, a different instance of the same method will be called
  • If several equal methods for different input conditions are called the same, the last one defined prevails
  • Basic overloading
									class Animal:
    def make_sound(self):
        return "Some generic sound"

class Dog(Animal):
    def make_sound(self):
        return "Woof!"

class Cat(Animal):
    def make_sound(self):
        return "Meow!"

# Function demonstrating dynamic polymorphism
def animal_sound(animal):
    return animal.make_sound()

my_animals = [Dog(), Cat()]

for animal in my_animals:
    print(animal_sound(animal))  # The decision is made at runtime

Vehicle interface

Use of superclasses that are not directly called or instanced, but provide a clear structure of the methods and attributes to be included on its subclasses. Learned:


  • Superclasses, subclasses and basic interface relations
  • Simple @abstractmethod decorator applications to force error raise if interface not followed

The Car subclass is not considering the stop method defined in the abstract class, therefore it will throw an error. Note the abc import, really handy (not compulsory) to structure abstract classes. The abc module defines the interface to which adopt our code. This ABC is used to define a class as abstract and @abstractmethod is a decorator. Note how the abstract methods START and STOP are overridden in the subclass: 

									from abc import ABC, abstractmethod

# Abstract class
class Vehicle(ABC):
    def __init__(self, make, model):
        self.make = make
        self.model = model

    @abstractmethod
    def start(self):
        pass

    @abstractmethod
    def stop(self):
        pass

# Subclasses inheriting from the abstract class
class Car(Vehicle):
    def start(self):
        return f"Starting the {self.make} {self.model} car"
    
class Motorcycle(Vehicle):
    def start(self):
        return f"Starting the {self.make} {self.model} motorcycle"

    def stop(self):
        return f"Stopping the {self.make} {self.model} motorcycle"

# Creating instances of the subclasses
my_car = Car("Toyota", "Corolla")
my_motorcycle = Motorcycle("Honda", "CBR")

Person details

A person's name is a constant value, unchangeable from the moment it is initialized. However, a person's age does change. Learned:


  • @getter and @setter methods
  • Manual getter and setter type of functions (def get_type(self) and def set_type(self) respectively)
  • Privacy Python convention, and strictly private convention for attributes not to be accessed or modified
									class Person:
    def __init__(self, name, age):
        self._name = name
        self._age = age

    @property
    def name(self):
        return self._name

    @property
    def age(self):
        return self._age

    @age.setter
    def age(self, new_age):
        if self._age > 20:
            raise ValueError("Age must be less than 20")
        self._age = new_age


c = Person("Calvin",23)
print(c.name)
print(c.age)
c.age = 56
print(c.age)
